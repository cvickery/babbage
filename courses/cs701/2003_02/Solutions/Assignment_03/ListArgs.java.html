      //  Class ListArgs
      //  ------------------------------------------------------------------
      /**
        *     Invokes a native method, doMain(), to print the command line
        *     arguments to this Java application.
        *
        *     @author C. Vickery
        *     Spring 2003
        */
        public class ListArgs
        {
          //  Static initialization
          //  --------------------------------------------------------------
          /*
           *    Loads shared library containing native method(s).
           */
            static
            {
              System.loadLibrary( "la" );
            }
      
          //  doMain()
          //  --------------------------------------------------------------
          /**
            *   Native method prints an array of strings.
            *
            *   @param  s   Array of strings to be printed.
            */
            public static native void doMain( String[] s );
      
          //  main()
          //  --------------------------------------------------------------
          /**
            *   Entry point for the application which passes command line
            *   arguments to a native method for printing, then prints them
            *   out itself.
            */
            public static void main( String[] args )
            {
              new ListArgs().doMain( args );
              for (int i = 0; i < args.length; i++)
              {
                System.out.println( "main: " + i + ": \t" + args[i] );
              }
              System.exit( 0 );
            }
        }
