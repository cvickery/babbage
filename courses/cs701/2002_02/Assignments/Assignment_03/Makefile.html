      CXXFLAGS    = -g -Wall -Wwrite-strings $(DEBUG_FLAG)
      LDFLAGS     =
      
      EXECS       = qserver qclient
      
      CLIENT_HDRS = qclient.h
      CLIENT_SRCS = qclient.cc logfile.cc
      CLIENT_OBJS = $(CLIENT_SRCS:.cc=.o)
      
      SERVER_HDRS = qserver.h
      SERVER_SRCS = qserver.cc logfile.cc
      SERVER_OBJS = $(SERVER_SRCS:.cc=.o)
      
      HDRS        = $(SERVER_HDRS) $(CLIENT_HDRS)
      SRCS        = $(SERVER_SRCS) $(CLIENT_SRCS)
      OBJS        = $(SERVER_OBJS) $(CLIENT_OBJS)
      
      default : $(EXECS)
      
      qserver : $(SERVER_OBJS)
      qclient : $(CLIENT_OBJS)
      
      clean   :
      	rcsclean
      	rm -f *.o *.bak *.log core $(EXECS)
      
      depend  : $(HDRS) $(SRCS)
      	makedepend -- $(CXXFLAGS) -- $(SRCS)
