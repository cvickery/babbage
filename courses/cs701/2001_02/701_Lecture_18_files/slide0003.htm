Creating/Overwriting a File
  #include <fcntl.h>
  #include <sys/stat.h>
  #include <sys/types.h>
  int open (
          const char *path,
          int oflag [ ,
          mode_t mode ]      );
  int creat (
          const char *path,
          mode_t mode );
 The following two function calls are equivalent:
    creat(path, mode);
    open(path, O_WRONLY | O_CREAT | O_TRUNC, mode);
