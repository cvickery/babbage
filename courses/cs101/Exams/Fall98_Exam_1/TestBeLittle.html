    public static void main(String[] args) {

    char[][] theArray = { { 'a', 'b', 'c', 'd' },
                          { 'Z', 'Y', 'W', 'V' },

                        };
      System.out.println( "The array is " + 100 * beLittle(theArray) +

  //  beLittle()
  //  -----------------------------------------------------------------
  /**
    *   The method to be written for the exam.
    *
    */
    
    static double beLittle(char[][] x) {

    double    numLittle = 0, total = 0;

      for (int row = 0; row < x.length; row++) {

        total += x[row].length;

        for (int col = 0; col < x[row].length; col++) {

          if (Character\.isLowerCase(x[row][col])) {

            numLittle++;

            }

          }

        }

      if ( total > 0 ) {

        return numLittle/total;

        }

      else {

        return 0;

        }
      }
    }
