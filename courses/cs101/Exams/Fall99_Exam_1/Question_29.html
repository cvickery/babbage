import javax.swing.*;

public class Question_29
{
  public static void main( String[] argv )
  {
    if ( argv.length == 0 )
    {
      JOptionPane.showMessageDialog( null,
                              "There were no command line arguments");
      System.exit( 0 );
    }
    for ( int i = 0; i < argv.length; i++ )
    {
      JOptionPane.showMessageDialog( null, argv[i] );
    }
    System.exit( 0 );
  }
}
