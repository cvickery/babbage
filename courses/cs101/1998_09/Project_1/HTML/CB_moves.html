/**
  *   Displays the valid moves for a board.
  *
  *   @author C. Vickery
  *
  */
  public class CB_moves extends CB_Processor {

    //  Constructor
    //  -----------
    public CB_moves() { setName( "moves" ); }

    //  process()
    //  --------------------------------------------------------------
    /**
      *   Processes moves commands by printing result of calling
      *   getMoves().
      *

        System.out.println( cb.getMoves() );

        }
