  //  Class CBException
  //  -----------------------------------------------------
  /**
    *   Exceptions thrown by Checkerboard methods
    *
    *   @author C. Vickery
    *
    */
  public class CBException extends Exception {
    private String message;

    //  Constructor
    //  ---------------------------------------------------
    CBException( String msg ) {
      message=msg;
      }

  //  toString()
  //  -----------------------------------------------------
  /**
    *   Returns the message.
    *
    */
    public String toString() {
      return message;
      }

    }
