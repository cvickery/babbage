//  CB_clear.java

import java.util.StringTokenizer;

//  Class CB_clear
//  -------------------------------------------------------------------
/**
  *   Removes all pieces from the board.
  *
  *   @author C. Vickery
  *
  */
  class CB_clear extends CB_Processor {

    //  Constructor
    //  -----------
    public CB_clear() { setName( "clear" ); }

    //  process()
    //  ---------
    /**
      *   Calls Checkerboard.zero() to clear the board.
      */
      public void process ( StringTokenizer st, Checkerboard cb ) {

      cb.zero();
      
      }

    }
