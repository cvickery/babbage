//  MouseControl_0.java

import java.awt.event.*;

//  Class MouseControl_0
//  -------------------------------------------------------------------
/**
  *   Prints the values of the InputEvent masks
  *   and MouseEvent constants.
  *
  *   @author C. Vickery
  *
  */
  public class MouseControl_0 {

    public static void main( String[] argv ) {

      System.out.println( "InputEvent Masks:" );
      System.out.println( " CTRL:\t"  + InputEvent.CTRL_MASK    );
      System.out.println( " SHIFT:\t" + InputEvent.SHIFT_MASK   );
      System.out.println( " ALT:\t"   + InputEvent.ALT_MASK     );
      System.out.println( " META:\t"  + InputEvent.META_MASK    );
      System.out.println( " BTN1:\t"  + InputEvent.BUTTON1_MASK );
      System.out.println( " BTN2:\t"  + InputEvent.BUTTON2_MASK );
      System.out.println( " BTN3:\t"  + InputEvent.BUTTON3_MASK );

      System.out.println( "MouseEvent Constants:" );
      System.out.println( " CLICKED:\t"  + MouseEvent.MOUSE_CLICKED  );
      System.out.println( " PRESSED:\t"  + MouseEvent.MOUSE_PRESSED  );
      System.out.println( " RELEASED:\t" + MouseEvent.MOUSE_RELEASED );
      System.out.println( " MOVED:   \t" + MouseEvent.MOUSE_MOVED    );
      System.out.println( " ENTERED:\t"  + MouseEvent.MOUSE_ENTERED  );
      System.out.println( " EXITED:\t"   + MouseEvent.MOUSE_EXITED   );
      System.out.println( " DRAGGED:\t"  + MouseEvent.MOUSE_DRAGGED  );

    }

  }
