// FrameControl_1.java

import java.awt.*;
import java.awt.event.*;

//  Class FrameControl_1
//  -------------------------------------------------------------------
/**
  *   Illustrates creating a Frame and using it to listen to its own
  *   window events
  *
  *   @author C. Vickery
  *
  */
  public class  FrameControl_1
                extends Frame
                implements WindowListener {

  //  Constructor
  //  -----------------------------------------------------------------
  /**
    *   Passes the title string to the superclass' constructor.
    *
    */
    FrameControl_1( String title ) {
      super( title);
    }

  //  main()
  //  -----------------------------------------------------------------
  /**
    *   Instantiates this class and establishes it as the Window
    *   Listener for this Frame.
    *
    */
    public static void main( String[] argv ) {

      FrameControl_1 fc = new FrameControl_1( "Frame Control 1" );
      fc.addWindowListener( fc );
      fc.setSize( 200, 200 );
      fc.show();
    }

  //  windowXXX()
  //  -----------------------------------------------------------------
  /**
    *   Displays a message telling when each of the seven types of
    *   Window events occurs.  All seven methods have to be defined
    *   even though windowClosing() is the only one that does anything
    *   because this class declares that it implements the
    *   WindowListener interface.
    *
    */
    public void windowActivated( WindowEvent we ) {
      System.out.println( "activated" );
    }

    public void windowClosed( WindowEvent we ) {
      System.out.println( "closed" );
    }

    public void windowClosing( WindowEvent we ) {
      System.out.println( "closing" );
      System.exit( 0 );
    }

    public void windowDeactivated( WindowEvent we ) {
      System.out.println( "deactivated" );
    }

    public void windowIconified( WindowEvent we ) {
      System.out.println( "iconified" );
    }

    public void windowDeiconified( WindowEvent we ) {
      System.out.println( "deiconified" );
    }

    public void windowOpened( WindowEvent we ) {
      System.out.println( "opened" );
    }

  }
