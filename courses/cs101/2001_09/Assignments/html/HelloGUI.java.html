      import javax.swing.*;
      
      /**  Prompts the user for his or her name, and displays a message
        *  in a window.
        *
        *   @author C. Vickery
        */
        public class HelloGUI
        {
          /***********************************************************
           *Prompts the user to enter his or her name in a dialog
           * box, then displays a message in another dialog box, and
           * exits.
           *
           *    @param args Command line arguments.  Not used in this
           *    program.
           *    @return Void.
           ***********************************************************/
          public static void main( String[] args )
          {
           String userName = JOptionPane.showInputDialog( null,
               "What is your name?" );
           JOptionPane.showMessageDialog( null, "Hello " + userName +
               ", My name is Dr. Vickery" );
           System.exit( 0 );
          }
        }
