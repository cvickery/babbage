import java.awt.*;

public class ShowCoordinates {
  public static void main( String[] argv ) {
    LineCoordinates lc = new LineCoordinates( 10, 10, 15, 9 );
    while ( lc.hasMorePoints() ) {
      Point p = lc.nextPoint();
      System.out.println( "x=" + p.x + ", y=" + p.y );
    }
  }
}
