import java.awt.*;
import java.awt.event.*;

//  Class FrameControl_3
//  -------------------------------------------------------------------
/**
  *   Uses a WindowAdapter to manage a Frame.
  *   There are separate Frame and WindowAdapter classes.
  *
  *   @author C. Vickery
  *
  */
  public class  FrameControl_2 extends Frame {

  //  Constructor
  //  -----------------------------------------------------------------
  /**
    *   Passes the title string to the superclass constructor.
    *
    */
    FrameControl_3( String title ) {
      super( title );
    }

  //  main()
  //  -----------------------------------------------------------------
  /**
    *   Instantiates this class, creating an object that is a subclass
    *   of Frame, and creates a separate object to provide a
    *   WindowAdapter for this Frame.
    *
    */
    public static void main( String[] argv ) {

    FrameControl_2 fc = new FrameControl_2( "Frame Control 2" );

      fc.addWindowListener( new FrameControl() );
      fc.setSize( 200, 200 );
      fc.show();
      // Demonstrate that this thread is not used any more.
      Thread.currentThread().stop();
      System.out.println( "If this prints, I'll give you $1M");

    }

  }


//  Class FrameControl
//  -------------------------------------------------------------------
/**
  *   Overrides the windowClosing method so that closing a Frame exits
  *   the application.
  *
  *   @author C. Vickery
  *
  */
  class FrameControl extends WindowAdapter {

  //  windowClosing()
  //  -----------------------------------------------------------------
  /**
    *   Exits the application.
    *
    */
    public void windowClosing( WindowEvent we ) {
      System.exit( 0 );
    }

  }
