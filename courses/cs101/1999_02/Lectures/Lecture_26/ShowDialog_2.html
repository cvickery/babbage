import javax.swing.*;

public class ShowDialog_2 {

  public static void main( String[] aboutMessages ) {

    if ( aboutMessages.length > 0 ) {
      JOptionPane.showMessageDialog(  null,
                                      aboutMessages,
                                      "Nice Messages",
                                      JOptionPane.PLAIN_MESSAGE );
    }

  }

}
