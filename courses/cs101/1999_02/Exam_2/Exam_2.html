import java.util.*;

public class Exam_2 {
  protected static final String NL = System.getProperty( "line.separator" );
  protected Vector vec;
  public Exam_2() { vec = new Vector(); }
  public void addString(String str) { vec.addElement( str ); }
}

class DoExam_2 extends Exam_2 {

  public static void main(String[] argv) {
    Exam_2 e2 = new DoExam_2();
    for (int i=0; i<argv.length; i++)
      e2.addString(argv[i]);
    System.out.print( e2 );
  }

  public String toString() {
    StringBuffer sb = new StringBuffer();
    for (int i = 0; i < vec.size(); i++)
      sb.append( vec.elementAt(i) + NL );
    return sb.toString();
  }

}
