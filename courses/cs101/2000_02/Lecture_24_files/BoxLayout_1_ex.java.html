import java.awt.*;
import javax.swing.*;

public class BoxLayout_1_ex
{
  public static void main( String[] argv )
  {
    JFrame jf = new JFrame( "BoxLayout_1_ex" );
    jf.setDefaultCloseOperation( JFrame.EXIT_ON_CLOSE );
    Container cp = jf.getContentPane();
    Box b = new Box( BoxLayout.X_AXIS );

    cp.add( b, BorderLayout.CENTER );

    b.add( new JButton( "First" ) );
    b.add( new JButton( "Second" ) );
    b.add( new JButton( "Third" ) );
    b.add( new JButton( "Fourth" ) );
    b.add( new JButton( "Fifth" ) );

    jf.pack();
    jf.setLocation( 200, 150 );
    jf.show();
  }

}
