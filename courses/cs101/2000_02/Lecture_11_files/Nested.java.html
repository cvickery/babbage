public class Nested
{
  //  Method Main()
  //  ---------------------------------------------------------
  public static void main( String[] argv )
  {

  int     myDepth = 0;
  String  myName  = "main";

    System.out.println( myDepth + ": " + myName );
    methodA( myName, myDepth );
    System.out.println( myDepth + ": Returned to " + myName);
    System.exit( 0 );
  }

  //  Method methodA()
  //  ---------------------------------------------------------
  private static void methodA( String called, int callerDepth )
  {
  int     myDepth = callerDepth + 1;
  String  myName  = "methodA";

    System.out.println( myDepth + ": " + myName );
    methodB( myName, myDepth );
    System.out.println( myDepth + ": Returned to " + myName);
    return;
  }

  //  Method methodB()
  //  ---------------------------------------------------------
  private static void methodB( String called, int callerDepth )
  {
  int     myDepth = callerDepth + 1;
  String  myName  = "methodB";
    System.out.println( myDepth + ": " + myName );
    methodC( myName, myDepth );
    System.out.println( myDepth + ": Returned to " + myName);
    return;
  }

  //  Method methodC()
  //  ---------------------------------------------------------
  private static void methodC( String called, int callerDepth )
  {
  int     myDepth = ++callerDepth;
  String  myName  = "methodC";
    System.out.println( myDepth + ": " + myName );
    return;
  }

}
