Accessor Methods
(JavaBeans Naming)
public class AccessorMethods {
  private int secretVar;
  private int readonlyVar;
  private int publicVar;
  private boolean setupComplete;
  public int getReadonlyVar()              { return readonlyVar;   }
  public int getPublicVar()                    { return publicVar;       }
  public void setPublicVar( int val )      { publicVar = val;         }
  public boolean isSetupComplete() { return setupComplete; }
// Other code inside the class can access all these variables directly.
}
