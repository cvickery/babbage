/**
  *   Holds a word as a String and a counter as an int.
  *
  *     @author   C. Vickery
  *     @version  1.0 - February, 2000
  */
  public class WordCounter
  {
    /** The String that holds the word.               */
    public String word;
    /** The int that holds the counter for a word. */
    public int    counter = 1;


  //  Constructor
  //  ----------------------------------------------------------------
  /**
    *   Initializes word with str, and sets counter to 1.
    *
    *     @param  str - The String to be saved as word.
    */
     public WordCounter( String str )
      {
        word = str.toLowerCase();
      }
  }
