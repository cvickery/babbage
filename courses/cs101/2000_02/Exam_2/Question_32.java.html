public class Question_32
{
  public static void main( String[] argv )
  {
    if ( argv.length != 1 )
    {
      System.err.println( "There must be exactly one argument." );
      System.exit( 1 );
    }
    if ( argv[0].endsWith( ".java" ) )
    {
      System.exit( 0 );
    }
    System.err.println( argv[0] + " does not end with \".java\"." );
    System.exit( 1 );
  }
}
