//  rest of the code in this file is so we can check that the method
//  actually works.

import java.io.*;

public class Question_31
{
  public static void copy( File f1, File f2 ) throws IOException
  {
    byte[]  ioBuf = new byte[1024];
    FileInputStream fis = new FileInputStream( f1 );
    FileOutputStream fos = new FileOutputStream( f2 );
    int numBytes = fis.read( ioBuf );
    while ( numBytes > 0 )
    {
      fos.write( ioBuf, 0, numBytes );
      numBytes = fis.read( ioBuf );
    }
    fos.close();
  }

  public static void main( String[] argv ) throws IOException
  {
    copy( new File(argv[0]), new File(argv[1]) );
    System.exit( 0 );
  }
}
