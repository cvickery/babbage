class Simple_Array_1D_a {

  public static void main( String[] argv ) {

  //  Declare an array of int and provide a list of initial values.
  int[] anArray = { 11, -33, 5, 0, -2000000000 };

    //  Print the value of the array's length field.
    System.out.println( "Length:\t\t"   + anArray.length );

    //  Iterate through the elements of the array, printing each
    //  subscript and the corresponding element's value.
    System.out.println( "\nSubscript\tValue" );
    for ( int i = 0; i < anArray.length; i++ ) {
      System.out.println( i + ":\t\t" + anArray[ i ] );
    }

    System.exit( 0 );

  }
}
