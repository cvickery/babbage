class Simple_Array_1D_b {

  public static void main( String[] argv ) {

  int[] anArray = { 11, -33, 5, 0, -2000000000 };

    System.out.println( "Length:\t\t"   + anArray.length );
    System.out.println( "\nSubscript\tValue" );

    //  Generate an ArrayIndexOutOfBoundsException.
    for ( int i = 0; i <= anArray.length; i++ ) {
      System.out.println( i + ":\t\t" + anArray[ i ] );
    }

    System.exit( 0 );

  }
}
