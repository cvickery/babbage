    /**
      *   Changes the genre using an index into genreNames.
      *
      *   @param  n     The index of the genre name to use.
      *   @return void
      *   @throws InvalidGenreNumberException
      *                 if n is not between 0 and genreNames.length.
      */
      public void setGenre( int n ) throws InvalidGenreNumberException
      {
        if ( (n < 0) || (n >= genreNames.length) )
        {
          throw new InvalidGenreNumberException(
                                          new Integer( n ).toString() );
        }
        genreCode  = n;
        genre = genreNames[ n ];
        infoModified = true;
      }
