    public double average( double[][] x ) {


    double sum = 0.0;

    int n = 0;

    for ( int row = 0; row < x.length; row++ ) {

    n += x[row].length;

    for ( int col = 0; col < x[row].length; col++)

    sum += x[row][col];

    }

    return sum / n;

    }
