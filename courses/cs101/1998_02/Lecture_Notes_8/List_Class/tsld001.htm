    abstract public class List {
    protected Object[] theList;
    private boolean isFull, isEmpty;
    private int front, back, capacity,
    numElements;
    protected List( int cap ) {
    theList = new Object[ cap ];
    capacity = cap;
    numElements = 0;
    front = 0;
    back = 0;
    isFull = false;
    isEmpty = true;
    }
