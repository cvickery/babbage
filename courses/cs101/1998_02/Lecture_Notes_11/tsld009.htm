    public void init() {


    if ( isApplet ) {

    add( quit );

    quit.addActionListener( this );

    }

    wff = new WF_Frame( "WordFind Puzzle",

    isApplet ? this : null );

    wff.setSize( 300, 200 );

    wff.show();

    }

    public void actionPerformed( ActionEvent e ) {

    wff.setVisible( ! wff.isVisible() );

    add( new Label( "Cannot exit an applet" ) );

    repaint();

    }

    }
