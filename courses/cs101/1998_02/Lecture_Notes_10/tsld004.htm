The Answer Class

    public class Answer {


    int row, col;

    Direction direction;

    // Constructor

    // ---------------------------------------------------------------

    public Answer( int r, int c, int d ) {

    row = r;

    col = c;

    direction = new Direction( d );

    }

    // Answer.toString()

    // ---------------------------------------------------------------

    public String toString() {

    return "row " + row + ", col " + col + ", " + direction;

    }

    }
